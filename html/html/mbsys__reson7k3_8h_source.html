24 /*
25  * mbsys_reson7k3.h defines the MBIO data structures for handling data from
26  * Teledyne Reson 7k series, Teledyne Odom MB2, Teledyne BlueView ProScan
27  * software, Hydrosweep 3rd generation (HS3) sonars and other applications
28  * and sonars using 7k data record formats:
29  * MBF_RESON7K3 : MBIO ID 89 - Teledyne Reson 3rd generation 7K data format
30  *
31  * Authors: C. S. Ferreira & D. W. Caress
32  * Date: March 2019
33  *
34  * Notes on the mbsys_reson7k3 data structure:
35  * 1. This format is defined by the 7k Data Format Definition (DFD)
36  * document for Teledyne RESON SeaBat 7k Data Format Version 3.10.
37  * 2. Reson 7k series multibeam sonars output bathymetry, per beam
38  * amplitude, sidescan data and water column.
39  * 3. Reson 7k format is used also to log sidescan and subbottom
40  * data from other sonars.
41  * 4. The 7k record consists of a data record frame (header and checksum),
42  * a record type header, an optional record data field and an optional
43  * data field for extra information. The optional data field typically
44  * holds sensor specific data and third party developer
45  * embedded data.
46  * 5. The format specification includes a number of data record types
47  * that have been part of the data format in the past but are now
48  * deprecated. Data including deprecated records should be read using
49  * the MBF_RESON7KR (88) i/o module. The data supported by this
50  * MBF_RESON7K3 (89) i/o module is expected to use the following
51  * data records for each ping:
52  * Sonar Settings
53  * 7000 – Sonar Settings
54  * Receiver Beam Geometry
55  * 7004 – Beam Geometry
56  * Bathymetry
57  * 7027 – Raw Detections (+ optional calculated bathymetry)
58  * 7047 – Segmented Raw Detections (+ optional calculated bathymetry)
59  * Sidescan time series
60  * 7007 – Side-scan data
61  * Backscatter samples (snippets)
62  * 7028 – Snippet
63  * Water column beamformed time series
64  * 7018 – Beamformed data
65  * 7041 – Compressed beamformed intensity data
66  * 7042 – Compressed Water Column data
67  * 6. MB-System stores beamflags used for bathymetry editing in the
68  * RawDetection quality values for each beam. The 7k 3.10 DFD
69  * defines use of only the first two bits of these 32 bit
70  * values. MB-System embeds the 8-bit beamflag values in bits 24-31.
71  * 7. The maximum dimensions defined below
72  * #define MBSYS_RESON7K_MAX_BEAMS 512
73  * #define MBSYS_RESON7K_MAX_SOUNDINGS 2560
74  * #define MBSYS_RESON7K_MAX_SEGMENTS 3
75  * #define MBSYS_RESON7K_MAX_PIXELS 4096
76  * reflect the various Teledyne mapping sonars.
77  * The Reson SeaBat T50 multibeams have a maximum of 512 formed beams
78  * and can produce up to five multi-pick soundings per beam, resulting
79  * in the maximum soundings of 2560. These are reported in 7027 Raw Detections
80  * records. The Hydrosweep MD/30 multibeam produces 320 formed beams with
81  * three soundings per beam produced by split beam processing for 960
82  * soundings. These are reported in the 7047 Segmented Raw Detections
83  * records.
84  * ?. Questions/decisions to be resolved before this i/o module is
85  * finalized:
86  * - The beamflag can be embedded in either the quality value or the
87  * flags value - we've chosen the quality value - is this best?
88  * - It's clear that the RawDetection optional data exists when the
89  * optional data offset is nonzero in the RawDetection record header
90  * is nonzero. How does PDS set the the optional data identifier value?
91  * All the DFD states is that this identifier is user defined.
92  * - We want to augment the RawDetection optional data space with the
93  * usual MB-System laid-out pseudosidescan, but need to use a different
94  * optional data identifier than PDS so that the code knows this.
95  * - How complete are the attitude time series that can be constructed
96  * from the ping motion records? How do these relate to the data stored
97  * in the asynchronous RollPitchHeave, Attitude, CustomAttitude, and
98  * Heading records?
99  */

427 typedef struct s7k3_header_struct {
428  u16 Version; // Version - Version of this frame (e.g.: 1, 2 etc.)
429  u16 Offset; // Offset - Offset in bytes from the start of the sync
430  // pattern to the start of the Record Type Header (RTH).
431  // This allows for expansion of the header
432  // whilst maintaining backward compatibility.
433  u32 SyncPattern; // Sync pattern - 0x0000FFFF
434  u32 Size; // Size - Size in bytes of this record from the start
435  // of the Protocol version field to the end of the
436  // checksum field - including any embedded data.
437  u32 OptionalDataOffset; // Data offset - Offset in bytes to optional data field from
438  // start of record. Zero implies no optional data.
439  u32 OptionalDataIdentifier; // User defined.
440  s7k3_time s7kTime; // 7KTIME u8*10 UTC - Time tag indicating when data was
441  // produced.
442  u16 RecordVersion; // Currently 1
443  u32 RecordType; // Record type - Identifier for record type of embedded data.
444  u32 DeviceId; // Device identifier - Identifier of the device to which
445  // this datum pertains.
446  u16 Reserved; // Reserved
447  u16 SystemEnumerator; // System enumerator - The enumerator is used to differentiate
448  // between devices with the same device identifiers in one
449  // installation/system.
450  u32 Reserved2; // Reserved
451  u16 Flags; // Flags - BIT FIELD:
452  // Bit 0 - Checksum
453  // 0 - invalid checksum
454  // 1 - valid checksum
455  // Bit 1-14 - Reserved (must be zero)
456  // Bit 15:
457  // 0 - Live data
458  // 1 - Recorded data
459  u16 Reserved3; // Reserved
460  u32 Reserved4; // Reserved
461  u32 FragmentedTotal; // Always zero
462  u32 FragmentNumber; // Always zero
463  // Following this header is:
464  // DATA SECTION Dynamic Record type specific data.
465  // Checksum u32 Sum of bytes in data section
466  // (optional, depends on bit 1 of Flags field).
467  // Note: the checksum field should be computed
468  // as a 64 bit unsigned integer with the least
469  // significant 32 bits used to populate this field
470  // thus ensuring a valid checksum and avoiding
471  // an explicit overflow.
472 } s7k3_header;
